//Implement Linear Queue ADT using an array. Implement Enqueue and Dequeue operation
#include<stdio.h>
#define n 10
int front=-1,rear=-1;
int queue[n];


void insert();
void delete();
void display();


int main(){
   int choice;
   do{
       printf("Enter the choice to perform on queue\n");
       printf("1)Insert\n2)Delete\n3)Display\n6)EXIT\n");
       scanf("%d",&choice);
           switch(choice){
               case 1: insert();
                       break;
               case 2: delete();
                       break;
               case 3: display();
                       break;
               case 4: printf("EXIT!!\n");
                       break;
               case 5: printf("Enter a valid choice\n");
                       break;
           }
   }while(choice != 6);
   return 0;
}


void insert(){
   int value;
   if(rear==n-1){
       printf("\nOVERFLOW!!!!\n");
   }


   else{
       if(front==-1){
       front=0;
       }
       printf("\nEnter a value to be inserted\n");
       scanf("%d",&value);
       rear++;
       queue[rear]=value;
   }
}


void delete(){
   int num;
   if(front==-1 || front > rear){
       printf("UNDERFLOW!!\n");
   }
   else{
       num=queue[front];
       front++;
   printf("The deleted item is %d\n",num);
   }
}


void display(){
   int i;
   if(front==-1 || front > rear ){
       printf("UNDERFLOW!!!\n");
   }
   else{
       for(i=front;i<=rear;i++){
           printf("\t %d",queue[i]);
       }
   printf("\n");
   }
}
